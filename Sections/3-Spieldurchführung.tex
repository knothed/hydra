\subsection{Trumpfkarte}
\rules{
\regel Abhängig vom Spieltyp sind spezielle Karten \textit{Trümpfe}. In absteigender Rangfolge sind dies folgende:
\begin{itemize}
	\item Beim gesunden Spiel, beim Antrag und beim Normalsolo: Die Standardtrümpfe (\ref{Standardtrümpfe}).
	\item Beim Farbsolo mit Trumpffarbe \textit{Trumpf}: Hydra, \textit{Trumpf-}Ass, -10, -König, -Dame, -Bube, -9.
	\item Beim Trumpfsolo mit Trumpfwert \textit{Wert}: Hydra, Kreuz-, Pik-, Herz-, Karo-\textit{Wert}.
\end{itemize}
\regel \label{Standardtrümpfe} Als \textit{Standardtrümpfe} werden, in absteigender Rangfolge, die folgenden Karten bezeichnet: Hydra, Herz-10, Kreuz-, Pik-, Herz-, Karo-Dame, Kreuz-, Pik-, Herz-, Karo-Bube, Karo-Ass.
}
